## 二、 重构后的主干伪代码 (供架构师审查)

```java
@Override
@Transactional(rollbackFor = Exception.class)
public R<Boolean> batchSaveSxfyttz(SxfyttzSaveReqDTO req, CurrentUser<InnerUser> currentUser) {
    // ================== 阶段一：绝对严密的前置校验 ==================
    if (CollUtil.isEmpty(req.getItemList())) {
        return R.fail("没有可保存的数据");
    }
    
    // 校验 mxId 非空及重复
    Set<Long> uniqueMxIds = new HashSet<>();
    for (SxfyttzSaveReqDTO.SxfyttzItemDTO item : req.getItemList()) {
        if (item.getMxId() == null) {
            return R.fail("提交的数据中存在缺失预提明细ID的记录，请检查");
        }
        if (!uniqueMxIds.add(item.getMxId())) {
            return R.fail("提交的数据中存在重复的预提明细，请检查");
        }
    }

    // 门禁校验：排名更新后不允许修改（作为强门禁，查询失败亦阻断）
    try {
        JdhzhQueryReqDTO hzQuery = buildSxfyttzJdhzhQuery(req.getJslx(), req.getJszq(), req.getSflb());
        if (hzQuery != null) {
            hzQuery.setCurrent(1L);
            hzQuery.setSize(1L);
            R<Page<JdhzhRespDTO>> hzResult = zbkhFeignClient.queryJdhzhList(hzQuery);
            if (hzResult != null && hzResult.getIsSuccess() && hzResult.getData() != null) {
                List<JdhzhRespDTO> hzRecords = hzResult.getData().getRecords();
                if (CollUtil.isNotEmpty(hzRecords)) {
                    String zt = hzRecords.get(0).getZt();
                    if (zt != null && zt.compareTo("03") >= 0) {
                        return R.fail("该考核周期已执行排名更新并计算金额，不允许再修改预提调整");
                    }
                }
            } else {
                return R.fail("校验汇总排名状态失败，不允许保存");
            }
        }
    } catch (Exception e) {
        log.error("调用考核服务校验状态异常", e);
        return R.fail("门禁校验异常，请稍后重试");
    }

    // ... 影像校验逻辑保持不变 ...

    // ================== 阶段二：查库构建 mxMap (权威快照) ==================
    List<Long> mxIds = new ArrayList<>(uniqueMxIds);
    List<SxfytMxDO> mxList = sxfytMxDmService.listByIds(mxIds);
    if (CollUtil.isEmpty(mxList) || mxList.size() != mxIds.size()) {
        return R.fail("部分预提明细在数据库中不存在或已失效，请刷新后重试");
    }
    Map<Long, SxfytMxDO> mxMap = mxList.stream().collect(Collectors.toMap(SxfytMxDO::getId, n -> n));

    // ================== 阶段三：一次遍历做计算与命令准备 (暂不写库) ==================
    BigDecimal totalYtqje = BigDecimal.ZERO;
    BigDecimal totalYthje = BigDecimal.ZERO;
    List<SxfyttzSaveReqDTO.SxfyttzItemDTO> tzItemList = new ArrayList<>();
    // 用于存放待执行的更新条件包装器
    List<LambdaUpdateWrapper<SxfytMxDO>> updateCommandList = new ArrayList<>();

    String operator = currentUser != null ? currentUser.getUsername() : "system";
    Date now = new Date();

    for (SxfyttzSaveReqDTO.SxfyttzItemDTO item : req.getItemList()) {
        SxfytMxDO currentMx = mxMap.get(item.getMxId());
        BigDecimal rawSxf = currentMx.getSxf(); // 数据库原始值
        BigDecimal currentSxf = rawSxf == null ? BigDecimal.ZERO : rawSxf; // Java 计算用
        BigDecimal ytqje = item.getYtqje() == null ? BigDecimal.ZERO : item.getYtqje();
        BigDecimal ythje = item.getYthje() == null ? BigDecimal.ZERO : item.getYthje();
        
        // 并发一致性保护！
        if (currentSxf.compareTo(ytqje) != 0) {
            return R.fail("预提明细金额已被他人修改，请刷新后重试");
        }

        totalYtqje = totalYtqje.add(ytqje);
        totalYthje = totalYthje.add(ythje);
        BigDecimal yttzce = ythje.subtract(ytqje);

        // 组装条件更新命令 (注意处理 null 语义)
        LambdaUpdateWrapper<SxfytMxDO> updateWrapper = new LambdaUpdateWrapper<>();
        updateWrapper.eq(SxfytMxDO::getId, item.getMxId());
        if (rawSxf == null) {
            updateWrapper.isNull(SxfytMxDO::getSxf);
        } else {
            updateWrapper.eq(SxfytMxDO::getSxf, rawSxf);
        }
        updateWrapper.set(SxfytMxDO::getSxf, ythje)
                     .set(SxfytMxDO::getYttzce, yttzce)
                     .set(SxfytMxDO::getUpdator, operator)
                     .set(SxfytMxDO::getUpdatedTime, now);
        updateCommandList.add(updateWrapper);

        // 仅当存在实际调整差额时，才进入记账集合
        if (yttzce.compareTo(BigDecimal.ZERO) != 0) {
            tzItemList.add(item);
        }
    }

    // ================== 阶段四：总额平衡校验 ==================
    if (totalYtqje.compareTo(totalYthje) != 0) {
        return R.fail("调整后预提金额合计(" + totalYthje + ")必须等于调整前预提金额合计(" + totalYtqje + ")");
    }

    // ================== 阶段五：执行条件更新 (写库阶段，失败必须抛异常) ==================
    for (LambdaUpdateWrapper<SxfytMxDO> wrapper : updateCommandList) {
        if (!sxfytMxDmService.update(wrapper)) {
            // 抛出异常触发 Spring 事务回滚
            throw new BizException(ResultErrorCodeEnum.FAIL.getCode(), "保存失败，数据已被并发修改或已失效");
        }
    }

    // ================== 阶段六：生成记账凭证 ==================
    if (CollUtil.isNotEmpty(tzItemList)) {
        generateSxfyttzPz(req, tzItemList, mxMap, currentUser);
    }

    return R.success(true);
}

/**
 * 手续费预提调整财务记账核心逻辑
 */
private void generateSxfyttzPz(SxfyttzSaveReqDTO req, List<SxfyttzSaveReqDTO.SxfyttzItemDTO> tzItemList, 
                               Map<Long, SxfytMxDO> mxMap, CurrentUser<InnerUser> currentUser) {
    // ... 前置获取主表业务类型、摘要等公共信息 (略) ...
    // 注意：主表的 jbjgbh, yhdm 等均从 mxMap 权威快照中取值
    
    List<CwjzYwlsMxDTO> list = new ArrayList<>();
    BigDecimal zjehj = BigDecimal.ZERO;

    for (SxfyttzSaveReqDTO.SxfyttzItemDTO item : tzItemList) {
        SxfytMxDO mx = mxMap.get(item.getMxId());
        BigDecimal currentSxf = mx.getSxf() == null ? BigDecimal.ZERO : mx.getSxf();
        BigDecimal ythje = item.getYthje() == null ? BigDecimal.ZERO : item.getYthje();
        BigDecimal tzce = ythje.subtract(currentSxf);

        // 二次防御：零差额绝对不记账
        if (tzce.compareTo(BigDecimal.ZERO) == 0) {
            continue;
        }

        CwjzYwlsMxDTO ytmx = new CwjzYwlsMxDTO();
        // 关键维度全部取自 mx 快照
        ytmx.setJbjgbh(mx.getJbjgbh());
        ytmx.setJbjgmc(mx.getJbjgmc());
        ytmx.setYhdm(mx.getYhdm());
        ytmx.setHsfl2(mx.getZhbh());
        // ... 设置其他通用流水字段 ...

        // 核心金额处理：明细 zje 必须保留 tzce 的正负符号！调增为正，调减为负。
        ytmx.setZje(tzce);
        ytmx.setJse(BigDecimal.ZERO);
        
        // 仅在统计绝对值总额时使用 abs()
        zjehj = zjehj.add(tzce.abs());
        
        list.add(ytmx);
    }

    // 防御：如果过滤后为空则直接返回
    if (CollUtil.isEmpty(list)) {
        return;
    }

    // ... 主表装配 ...
    cwjzYwlsDTO.setZje(BigDecimal.ZERO); 
    cwjzYwlsDTO.setJse(BigDecimal.ZERO);
    // 业务发生额记录单边总额，使用绝对值汇总的一半
    BigDecimal dbje = zjehj.divide(new BigDecimal("2"), 2, RoundingMode.HALF_UP);
    cwjzYwlsDTO.setYwfse(dbje);
    cwjzYwlsDTO.setCwjzYwlsMxDTOList(list);

    R<CwjzRespDTO> cwjzRespDTOR = receiveCwjzxx(cwjzYwlsDTO, null);
    if (!cwjzRespDTOR.getIsSuccess()) {
        // 抛出异常回滚前面的明细更新
        throw new BizException(ResultErrorCodeEnum.FAIL.getCode(), "手续费预提调整财务记账失败：" + cwjzRespDTOR.getMsg());
    }
}
```
